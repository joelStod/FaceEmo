# Calculate the disattenuated correlation
# First calculate reliabilty with 1-SE^2/SD^2, or the proportion of variance due to measure error.
# To get measurement error, square standard error and then average the standard errors to get se SE^2
rel_IP_post<-1-(mean(AnaDBComp$IP_post_se^2,na.rm=TRUE))/(sd(AnaDBComp$IP_post)^2)
rel_eIP_post<-1-(mean(AnaDBComp$eIP_post_se^2, na.rm=TRUE))/(sd(AnaDBComp$eIP_post)^2)

# Next calculate rho, where rho= r_xy/sqrt(r_xx*r_yy)
with(AnaDBComp,cor(eIP_post,IP_post)/sqrt(rel_IP_post*rel_eIP_post))
## [1] 0.9379319
# Disattenuated correlation = 0.9379319
